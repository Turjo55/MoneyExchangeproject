package com.example.group57moneyexchangeproject;

import java.io.*;
import java.util.ArrayList;
import java.util.List;

public class LeaveApplication implements Serializable {
    private static final long serialVersionUID = 1L;

    private String name;
    private String id;
    private Integer days;
    private String department;
    private String reason;
    private String status;

    public LeaveApplication(String name, String id, Integer days, String department, String reason, String status) {
        this.name = name;
        this.id = id;
        this.days = days;
        this.department = department;
        this.reason = reason;
        this.status = status;
    }


    public String getName() {
        return name;
    }

    public String getId() {
        return id;
    }

    public Integer getDays() {
        return days;
    }

    public String getDepartment() {
        return department;
    }

    public String getReason() {
        return reason;
    }

    public String getStatus() {
        return status;
    }

    public void setStatus(String status) {
        this.status = status;
    }


    public static void saveToFile(List<LeaveApplication> applications, String filePath) {
        try (ObjectOutputStream oos = new ObjectOutputStream(new FileOutputStream(filePath))) {
            oos.writeObject(applications);
        } catch (IOException e) {
            System.err.println("Error saving leave applications to file: " + e.getMessage());
        }
    }

    public static List<LeaveApplication> loadFromFile(String filePath) {
        try (ObjectInputStream ois = new ObjectInputStream(new FileInputStream(filePath))) {
            return (List<LeaveApplication>) ois.readObject();
        } catch (IOException | ClassNotFoundException e) {
            System.err.println("Error loading leave applications from file: " + e.getMessage());
            return new ArrayList<>();
        }
    }
}
